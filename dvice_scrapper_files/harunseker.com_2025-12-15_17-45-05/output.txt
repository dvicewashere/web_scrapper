  <script>
    const portfolioBtn = document.getElementById('portfolioBtn');
    const backBtn = document.getElementById('backBtn');
    const heroSection = document.getElementById('heroSection');
    const portfolioSection = document.getElementById('portfolioSection');
    const langToggle = document.getElementById('langToggle');

    // --- Dil Verileri ve Fonksiyonlarƒ± ---
    const translatableElements = document.querySelectorAll('[data-tr], [data-en]');
    const defaultLang = 'tr';
    
    // 1. Dil Y√ºkleme
    function loadLanguage() {
      // Yerel depolama kontrol√º, yoksa varsayƒ±lan dil 'tr'
      const currentLang = localStorage.getItem('language') || defaultLang;
      setLanguage(currentLang);
    }
    
    // 2. Dili Deƒüi≈ütirme ve Metinleri G√ºncelleme
    function setLanguage(lang) {
      localStorage.setItem('language', lang);
      langToggle.textContent = lang.toUpperCase(); // Buton metnini EN/TR olarak ayarla

      translatableElements.forEach(el => {
        const key = `data-${lang}`;
        const translation = el.getAttribute(key);
        
        if (translation !== null) {
             // Metin ƒ∞√ßeriƒüi G√ºncelleme
            if (el.tagName === 'A' || el.tagName === 'BUTTON' || el.tagName === 'P' || el.tagName === 'H2' || el.tagName === 'H3' || el.tagName === 'DIV') {
                el.textContent = translation;
            }
        }
      });
      
      // Typing efekti i√ßin √∂zel durum:
      const typingElement = document.querySelector('.typing');
      if(typingElement) {
        typingElement.style.width = '0';
        typingElement.textContent = typingElement.getAttribute(`data-${lang}`);
        typingElement.style.animation = 'none'; 
        void typingElement.offsetWidth; // Reflow zorla
        typingElement.style.animation = 'typing 4s steps(30, end) forwards, blink 0.8s infinite';
      }
    }
    
    // 3. Dil Deƒüi≈ütirme Olayƒ±
    langToggle.addEventListener('click', () => {
      const currentLang = localStorage.getItem('language') || defaultLang;
      const newLang = currentLang === 'tr' ? 'en' : 'tr';
      setLanguage(newLang);
    });

    // --- Sayfa Y√ºkleme Olaylarƒ± ---
    document.addEventListener('DOMContentLoaded', () => {
        loadLanguage(); // Dili y√ºkle
    });

    // --- Portf√∂y Ge√ßi≈ü Olaylarƒ± ---
    portfolioBtn.addEventListener('click', (e) => {
      e.preventDefault();
      heroSection.classList.add('fade-out');

      setTimeout(() => {
        portfolioSection.classList.add('active');
        portfolioSection.scrollTop = 0;
      }, 500);
    });

    backBtn.addEventListener('click', () => {
      portfolioSection.classList.remove('active');
      
      setTimeout(() => {
        heroSection.classList.remove('fade-out');
      }, 500);
    });
  </script>
